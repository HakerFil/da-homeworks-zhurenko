Задача 1
SELECT orderNumber, productCode, quantityOrdered * priceEach AS Stoimost
FROM orderdetails 
ORDER BY Stoimost DESC
LIMIT 10

Задача 2
SELECT orderNumber, SUM(quantityOrdered * priceEach) AS total
FROM orderdetails 
GROUP BY orderNumber
HAVING total > 59000
ORDER BY total DESC


Задача 3
SELECT orderdetails.orderNumber, SUM(quantityOrdered * priceEach) AS total, orderDate, status
FROM orderdetails
INNER JOIN orders
ON orderdetails.orderNumber = orders.orderNumber
GROUP BY orderNumber
HAVING total > 59000
ORDER BY total DESC




Задача 4
SELECT orderdetails.orderNumber, SUM(quantityOrdered * priceEach) AS total, orderDate, status, contactFirstName, contactLastName, country
FROM orderdetails
INNER JOIN orders
ON orderdetails.orderNumber = orders.orderNumber
INNER JOIN customers
ON customers.customerNumber = orders.customerNumber
GROUP BY orderNumber
HAVING total > 59000
ORDER BY total DESC

Задача 5
SELECT productName, SUM(quantityOrdered * priceEach) AS total
FROM orderdetails
INNER JOIN products
ON orderdetails.productCode = products.productCode
GROUP BY productName
ORDER BY total DESC
LIMIT 10

Задача 6
SELECT firstName, lastName, c1.contactFirstName, c1.contactLastName
FROM employees
LEFT JOIN customers c1
ON c1.salesRepEmployeeNumber = employees.employeeNumber
RIGHT JOIN customers c2
ON c2.salesRepEmployeeNumber = employees.employeeNumber
ORDER BY firstName DESC
LIMIT 10

Задача 7
SELECT e2.firstName, e2.lastName, e2.jobTitle, e1.firstName AS subFirstName, e1.lastName AS subLastName
FROM employees e2
LEFT JOIN employees e1
ON e2.employeeNumber = e1.reportsTo
